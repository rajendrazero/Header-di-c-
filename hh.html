Contoh Tabel Header C++

Header Sintaks Pengertian
cctype #include <cctype> Menyediakan fungsi-fungsi karakter, seperti isalpha, isdigit, toupper, dan tolower.
clocale #include <clocale> Menyediakan fasilitas untuk bekerja dengan pengaturan lokal (locale).
csetjmp #include <csetjmp> Menyediakan fasilitas untuk bekerja dengan titik-singgah (setjmp).
csignal #include <csignal> Menyediakan fasilitas untuk bekerja dengan sinyal (signal) dalam program.
cstddef #include <cstddef> Menyediakan definisi-definisi umum dan konstanta-konstanta.
cstring #include <cstring> Menyediakan fungsi-fungsi untuk bekerja dengan string dalam gaya C.
cwchar #include <cwchar> Menyediakan fasilitas untuk bekerja dengan string lebar (wide string).
cwctype #include <cwctype> Menyediakan fungsi-fungsi karakter lebar (wide character).
type_traits #include <type_traits> Menyediakan fasilitas untuk menguji sifat-sifat tipe.
tuple #include <tuple> Menyediakan kelas tuple dan fungsi-fungsi terkait untuk bekerja dengan tuple.
queue #include <queue> Menyediakan kelas queue untuk struktur data antrian (queue).
stack #include <stack> Menyediakan kelas stack untuk struktur data tumpukan (stack).
deque #include <deque> Menyediakan kelas deque untuk struktur data deque.
list #include <list> Menyediakan kelas list untuk struktur data linked list.
map #include <map> Menyediakan kelas map untuk struktur data map (associative array).
unordered_map #include <unordered_map> Menyediakan kelas unordered_map untuk struktur data hash map (associative array tanpa urutan).
set #include <set> Menyediakan kelas set untuk struktur data set (himmapan himpunan).
unordered_set #include <unordered_set> Menyediakan kelas unordered_set untuk struktur data hash set (himmapan himpunan tanpa urutan).
bitset #include <bitset> Menyediakan kelas bitset untuk representasi bit.
functional #include <functional> Menyediakan fungsi-fungsi objek dan kelas-kelas fungsi untuk pemrograman fungsional.
limits #include <limits> Menyediakan informasi tentang batas-batas tipe data.
memory #include <memory> Menyediakan fasilitas untuk alokasi dan dealokasi memori secara dinamis.
regex #include <regex> Menyediakan fasilitas untuk bekerja dengan ekspresi reguler (regular expressions).
typeindex #include <typeindex> Menyediakan fasilitas untuk mengidentifikasi tipe data pada waktu kompilasi.
typeinfo #include <typeinfo> Menyediakan fasilitas untuk informasi tipe data pada waktu runtime.
atomic #include <atomic> Menyediakan fasilitas untuk operasi-operasi atomik.
mutex #include <mutex> Menyediakan fasilitas untuk penguncian dan pembukaan kunci sumber daya dalam multithreading.
thread #include <thread> Menyediakan fasilitas untuk pembuatan dan pengendalian thread dalam multithreading.
future #include <future> Menyediakan fasilitas untuk mengelola hasil operasi asinkron (future) dalam multithreading.
condition_variable #include <condition_variable> Menyediakan fasilitas untuk menunggu dan memberi sinyal pada kondisi tertentu dalam multithreading.
random #include <random> Menyediakan fasilitas untuk generasi angka acak.
filesystem #include <filesystem> Menyediakan fasilitas untuk berinteraksi dengan sistem file.
numeric #include <numeric> Menyediakan fungsi-fungsi untuk operasi-operasi numerik pada rangkaian data.
valarray #include <valarray> Menyediakan kelas valarray untuk operasi-operasi numerik pada array nilai.
iterator #include <iterator> Menyediakan fasilitas untuk bekerja dengan iterator dalam STL (Standard Template Library).
string #include <string> Menyediakan kelas string untuk bekerja dengan string.
iomanip #include <iomanip> Menyediakan fungsi-fungsi untuk memanipulasi input/output (I/O) secara berformat.
iostream #include <iostream> Menyediakan fungsi-fungsi untuk input/output (I/O) dasar.
cstdlib #include <cstdlib> Menyediakan fungsi-fungsi utilitas umum, seperti pengelolaan memori, konversi angka, dan fungsi-fungsi lainnya.
cmath #include <cmath> Menyediakan fungsi-fungsi matematika dasar, seperti sqrt, pow, sin, cos, dan lain-lain.
ctime #include <ctime> Menyediakan fungsi-fungsi untuk bekerja dengan waktu dan tanggal.
cassert #include <cassert> Menyediakan fasilitas untuk pengujian asersi.
cstdio #include <cstdio> Menyediakan fungsi-fungsi untuk input/output (I/O) dalam gaya C.
